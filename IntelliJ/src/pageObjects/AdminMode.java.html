29               WebElement reorderToggle;
30   
31       public void clickOnAddNewCustomOption() {
32           wait.until(ExpectedConditions.elementToBeClickable(addNewCustomOption));  // Waits until the "Add New Custom Option" button is clickable.
33           JavascriptExecutor js = (JavascriptExecutor) driver;
34           js.executeScript("arguments[0].click();", addNewCustomOption);
35       }
